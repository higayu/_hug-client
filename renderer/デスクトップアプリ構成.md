# ファイル構成
```
renderer/
├── index.html              # メインHTML
├── style.css               # メインCSS
├── mainRenderer.js         # メインJavaScript
├── modules/                # モジュールフォルダ
│   ├── settingsEditor.js   # 設定エディター（パス更新済み）
│   └── ...
└── settings/               # 🆕 設定モーダル専用フォルダ
    ├── modal.html          # 設定モーダルのHTML
    ├── modal.css           # 設定モーダルのCSS
    ├── modal.js            # 設定モーダル専用JavaScript
    ├── config.js           # 設定モーダル用設定ファイル
    └── README.md           # ドキュメント
```
